# Plan 1.4: Campaign Worker + Actions

## Objective

Реалізувати серце автоматизації: template engine для персоналізації повідомлень, логіку виконання кроків кампанії з умовами, і головний campaign worker що обробляє черги лідів через cron кожні 5 хвилин. Після цього плану: система може виконувати кампанію в реальному часі.

## must_haves

```yaml
truths:
  - backend/src/utils/templates.ts реалізує підстановку {firstName}, {lastName}, {company}, {title}, {myName}
  - backend/src/services/campaign.ts реалізує processLead з умовами if_connected / if_not_replied
  - backend/src/workers/campaignWorker.ts запускається через node-cron кожні 5 хвилин
  - Worker перевіряє workingHours і dailyLimits перед кожною дією
  - Ліди зі статусом connection_sent перевіряються на acceptance (24h polling, 14-day skip)
  - Кожна зміна статусу ліда оновлюється в БД через storage.ts
```

```typescript
import { getSetting } from '../services/storage';

export interface TemplateContext {
  firstName: string;
  lastName: string;
  company: string;
  title: string;
  myName?: string;
}

// Підстановка змінних у шаблон
// Всі змінні у форматі {variableName}
export function renderTemplate(template: string, context: TemplateContext): string {
  const myName = context.myName || getSetting('my_name') || 'there';

  return template
    .replace(/\{firstName\}/gi, context.firstName)
    .replace(/\{lastName\}/gi, context.lastName)
    .replace(/\{company\}/gi, context.company)
    .replace(/\{title\}/gi, context.title)
    .replace(/\{myName\}/gi, myName);
}

// Валідація connection note (ліміт LinkedIn: 300 символів)
export function validateConnectionNote(note: string): { valid: boolean; error?: string } {
  if (note.length > 300) {
    return { valid: false, error: `Connection note too long: ${note.length}/300 characters` };
  }
  return { valid: true };
}

// Валідація повідомлення (ліміт LinkedIn: 1900 символів)
export function validateMessage(message: string): { valid: boolean; error?: string } {
  if (message.length > 1900) {
    return { valid: false, error: `Message too long: ${message.length}/1900 characters` };
  }
  return { valid: true };
}

// Отримати всі змінні що використовуються в шаблоні
export function getTemplateVariables(template: string): string[] {
  const matches = template.match(/\{(\w+)\}/g) || [];
  return [...new Set(matches.map(m => m.slice(1, -1)))];
}

// Превью шаблону для конкретного ліда
export function previewTemplate(template: string, context: TemplateContext): string {
  return renderTemplate(template, context);
}
```

```typescript
import { getDb, incrementDailyCounter } from './storage';
import { canPerformAction, isWithinWorkingHours } from '../utils/delays';
import { renderTemplate, validateConnectionNote, validateMessage } from '../utils/templates';
import { logAction } from '../utils/logger';
import {
  visitProfile,
  sendConnectionRequest,
  sendMessage,
  checkConnectionStatus,
} from './linkedin';
import { actionDelay } from '../utils/humanizer';

export interface CampaignStep {
  id: string;
  campaign_id: string;
  type: 'visit_profile' | 'send_connection' | 'send_message' | 'send_inmail' | 'wait';
  step_order: number;
  config: {
    message?: string;
    delayDays?: number;
    condition?: 'always' | 'if_connected' | 'if_not_replied';
  };
}

export interface Lead {
  id: string;
  campaign_id: string;
  first_name: string;
  last_name: string;
  company: string;
  title: string;
  profile_url: string;
  status: string;
  current_step: number;
  visited_at?: string;
  connection_sent_at?: string;
  connected_at?: string;
  message_sent_at?: string;
  replied_at?: string;
}

// Оновити статус ліда в БД
export function updateLeadStatus(
  leadId: string,
  status: string,
  stepIncrement = 0,
  extraFields: Record<string, string | null> = {}
): void {
  const db = getDb();
  const now = new Date().toISOString();
  const fields = ['status = ?', 'updated_at = ?'];
  const values: (string | number | null)[] = [status, now];

  if (stepIncrement > 0) {
    fields.push('current_step = current_step + ?');
    values.push(stepIncrement);
  }

  for (const [key, val] of Object.entries(extraFields)) {
    fields.push(`${key} = ?`);
    values.push(val);
  }

  values.push(leadId);
  db.prepare(`UPDATE leads SET ${fields.join(', ')} WHERE id = ?`).run(...values);
}

// Перевірити чи виконується умова кроку для ліда
function checkStepCondition(step: CampaignStep, lead: Lead): boolean {
  const condition = step.config.condition || 'always';

  switch (condition) {
    case 'always':
      return true;
    case 'if_connected':
      return lead.status === 'connected' || !!lead.connected_at;
    case 'if_not_replied':
      return !lead.replied_at;
    default:
      return true;
  }
}

// Виконати один крок для ліда
export async function executeStep(lead: Lead, step: CampaignStep): Promise<void> {
  const context = {
    firstName: lead.first_name,
    lastName: lead.last_name,
    company: lead.company,
    title: lead.title,
  };

  switch (step.type) {
    case 'visit_profile': {
      if (!canPerformAction('visit')) {
        throw new Error('Daily profile visit limit reached');
      }
      updateLeadStatus(lead.id, 'visiting');
      await visitProfile(lead.profile_url);
      await incrementDailyCounter('profiles_visited');
      await actionDelay();
      updateLeadStatus(lead.id, 'pending', 1, { visited_at: new Date().toISOString() });
      logAction({
        leadId: lead.id,
        action: 'visit_profile',
        message: `✅ Visited profile: ${lead.first_name} ${lead.last_name} (${lead.company})`,
      });
      break;
    }

    case 'send_connection': {
      if (!canPerformAction('connection')) {
        throw new Error('Daily connection request limit reached');
      }
      let note: string | undefined;
      if (step.config.message) {
        const rendered = renderTemplate(step.config.message, context);
        const validation = validateConnectionNote(rendered);
        if (!validation.valid) throw new Error(validation.error);
        note = rendered;
      }
      await sendConnectionRequest(lead.profile_url, note);
      await incrementDailyCounter('connections_sent');
      await actionDelay();
      updateLeadStatus(lead.id, 'connection_sent', 1, {
        connection_sent_at: new Date().toISOString(),
      });
      logAction({
        leadId: lead.id,
        action: 'send_connection',
        message: `📤 Connection request sent to ${lead.first_name} ${lead.last_name}`,
      });
      break;
    }

    case 'send_message': {
      if (!canPerformAction('message')) {
        throw new Error('Daily message limit reached');
      }
      if (!step.config.message) throw new Error('No message template configured for this step');
      const rendered = renderTemplate(step.config.message, context);
      const validation = validateMessage(rendered);
      if (!validation.valid) throw new Error(validation.error);

      await sendMessage(lead.profile_url, rendered);
      await incrementDailyCounter('messages_sent');
      await actionDelay();
      updateLeadStatus(lead.id, 'message_sent', 1, {
        message_sent_at: new Date().toISOString(),
      });
      logAction({
        leadId: lead.id,
        action: 'send_message',
        message: `💬 Message sent to ${lead.first_name} ${lead.last_name}`,
      });
      break;
    }

    case 'wait': {
      // Просто перевірити чи пройшов потрібний час
      const delayDays = step.config.delayDays || 1;
      const lastActionDate = lead.connection_sent_at || lead.visited_at || lead.message_sent_at;
      if (!lastActionDate) {
        updateLeadStatus(lead.id, 'pending', 1);
        break;
      }
      const daysPassed = (Date.now() - new Date(lastActionDate).getTime()) / (1000 * 60 * 60 * 24);
      if (daysPassed >= delayDays) {
        updateLeadStatus(lead.id, 'pending', 1);
      }
      // Інакше — лід залишається на цьому кроці
      break;
    }

    default:
      logAction({
        leadId: lead.id,
        action: step.type,
        message: `⚠️ Unknown step type: ${step.type}`,
        level: 'warn',
      });
  }
}

// Обробити одного ліда в кампанії
export async function processLead(lead: Lead): Promise<void> {
  const db = getDb();
  const steps = db.prepare(
    'SELECT * FROM campaign_steps WHERE campaign_id = ? ORDER BY step_order ASC'
  ).all(lead.campaign_id) as CampaignStep[];

  if (lead.current_step >= steps.length) {
    updateLeadStatus(lead.id, 'completed');
    return;
  }

  const step = steps[lead.current_step];
  if (!checkStepCondition(step, lead)) {
    // Умова не виконується — пропустити крок
    updateLeadStatus(lead.id, lead.status, 1);
    return;
  }

  await executeStep(lead, step);
}

// Перевірити ліди що чекають на acceptance connection request
export async function checkPendingConnections(): Promise<void> {
  const db = getDb();
  const pendingLeads = db.prepare(
    "SELECT * FROM leads WHERE status = 'connection_sent' AND connection_sent_at IS NOT NULL"
  ).all() as Lead[];

  for (const lead of pendingLeads) {
    const daysSinceRequest = (Date.now() - new Date(lead.connection_sent_at!).getTime()) / (1000 * 60 * 60 * 24);

    // Якщо минуло 14 днів — пропустити
    if (daysSinceRequest > 14) {
      updateLeadStatus(lead.id, 'skipped');
      logAction({
        leadId: lead.id,
        action: 'check_connection',
        message: `⏭ Skipped ${lead.first_name} ${lead.last_name} — 14 days without acceptance`,
        level: 'warn',
      });
      continue;
    }

    // Якщо пройшло менше 24 годин — не перевіряти
    if (daysSinceRequest < 1) continue;

    const status = await checkConnectionStatus(lead.profile_url);
    if (status === 'connected') {
      updateLeadStatus(lead.id, 'connected', 0, { connected_at: new Date().toISOString() });
      logAction({
        leadId: lead.id,
        action: 'check_connection',
        message: `🤝 ${lead.first_name} ${lead.last_name} accepted connection!`,
        level: 'info',
      });
    }
  }
}
```

```typescript
import cron from 'node-cron';
import { getDb } from '../services/storage';
import { processLead, checkPendingConnections, Lead } from '../services/campaign';
import { isWithinWorkingHours, getDailyLimits, SAFE_LIMITS } from '../utils/delays';
import { leadDelay } from '../utils/humanizer';
import { logAction, logger } from '../utils/logger';

let isRunning = false;

async function runCampaignCycle(): Promise<void> {
  if (isRunning) {
    logger.warn('Campaign worker already running, skipping this cycle');
    return;
  }

  isRunning = true;

  try {
    // Перевірити робочий час
    if (!isWithinWorkingHours()) {
      return; // Тихо виходимо — не логуємо кожні 5 хвилин
    }

    const db = getDb();

    // 1. Перевірити ліди що чекають на прийняття запиту
    await checkPendingConnections();

    // 2. Отримати всі активні кампанії
    const activeCampaigns = db.prepare(
      "SELECT * FROM campaigns WHERE status = 'active'"
    ).all() as Array<{ id: string; name: string; settings: string }>;

    for (const campaign of activeCampaigns) {
      const settings = JSON.parse(campaign.settings) as {
        workingHours?: { start: number; end: number };
        workingDays?: number[];
        timezone?: string;
      };
      const timezone = settings.timezone || 'America/New_York';

      // Перевірити робочий час специфічний для кампанії
      if (!isWithinWorkingHours(timezone)) continue;

      // Перевірити денні ліміти
      const limits = getDailyLimits();
      if (!limits.canSendConnection && !limits.canSendMessage && !limits.canVisitProfile) {
        logAction({
          action: 'limit_reached',
          message: `⏸ Daily limits reached. Pausing until tomorrow ${SAFE_LIMITS.workingHours.start}:00`,
          campaignId: campaign.id,
        });
        continue;
      }

      // Отримати наступного ліда для обробки
      const nextLead = db.prepare(
        "SELECT * FROM leads WHERE campaign_id = ? AND status IN ('pending') ORDER BY created_at ASC LIMIT 1"
      ).get(campaign.id) as Lead | undefined;

      if (!nextLead) continue;

      try {
        await processLead(nextLead);
        // Затримка між лідами (1-3 хв)
        await leadDelay();
      } catch (err) {
        const errorMsg = String(err);

        // Детектуємо LinkedIn попередження
        if (errorMsg.includes('captcha') || errorMsg.includes('restriction') ||
            errorMsg.includes('checkpoint') || errorMsg.includes('limit')) {
          // Зупинити кампанію при серйозному попередженні
          db.prepare("UPDATE campaigns SET status = 'error', updated_at = ? WHERE id = ?")
            .run(new Date().toISOString(), campaign.id);
          logAction({
            action: 'warning',
            message: `🔴 Campaign "${campaign.name}" stopped: ${errorMsg}`,
            campaignId: campaign.id,
            level: 'error',
          });
        } else {
          // Позначити ліда як error і продовжувати
          db.prepare(
            "UPDATE leads SET status = 'error', error_message = ?, updated_at = ? WHERE id = ?"
          ).run(errorMsg, new Date().toISOString(), nextLead.id);
          logAction({
            action: 'warning',
            message: `⚠️ Lead ${nextLead.first_name} ${nextLead.last_name} error: ${errorMsg}`,
            campaignId: campaign.id,
            leadId: nextLead.id,
            level: 'warn',
          });
        }
      }
    }
  } finally {
    isRunning = false;
  }
}

// Запускати кожні 5 хвилин
export function startCampaignWorker(): void {
  cron.schedule('*/5 * * * *', async () => {
    await runCampaignCycle();
  });

  logger.info('Campaign worker started — running every 5 minutes');
}

// Зупинити всі активні кампанії (emergency pause)
export function pauseAllCampaigns(): void {
  const db = getDb();
  const count = db.prepare(
    "UPDATE campaigns SET status = 'paused', updated_at = ? WHERE status = 'active'"
  ).run(new Date().toISOString()).changes;
  logger.warn(`Emergency pause: ${count} campaigns paused`);
}
```

## Notes

- `isRunning` guard запобігає паралельному запуску worker якщо попередній цикл ще не завершився
- При помилці з 'captcha'/'restriction'/'checkpoint' — кампанія зупиняється, при звичайній помилці — лід позначається error і кампанія продовжується
- `pauseAllCampaigns` — для кнопки emergency pause на дашборді
- Worker не логує "not working hours" щоб не засмічувати logs вночі
