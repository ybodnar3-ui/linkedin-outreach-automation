# Plan 1.5: REST API Routes

## Objective

Реалізувати повний REST API: CRUD для кампаній, лідів, аналітика і налаштування. Включно з CSV-імпортом лідів (validation + deduplification), підключенням LinkedIn акаунта через API, і всіма статусними ендпоінтами.

## must_haves

```yaml
truths:
  - GET /api/campaigns повертає JSON масив кампаній (200)
  - POST /api/campaigns/id/start змінює status на 'active'
  - POST /api/leads/import приймає multipart/form-data з файлом CSV і повертає { added, skipped, errors }
  - POST /api/settings/login запускає startManualLogin() і повертає { status: 'login_started' }
  - GET /api/settings/session повертає { status: 'active' | 'expired' | 'not_connected' }
  - Всі routes підключені в index.ts
```

import { Router, Request, Response } from 'express';
import { v4 as uuidv4 } from 'uuid';
import { getDb } from '../services/storage';
import { pauseAllCampaigns } from '../workers/campaignWorker';

export const campaignsRouter = Router();

// GET /api/campaigns
campaignsRouter.get('/', (_req: Request, res: Response) => {
  const campaigns = getDb().prepare(`
    SELECT c.*, 
      (SELECT COUNT(*) FROM leads WHERE campaign_id = c.id) as total_leads
    FROM campaigns c
    ORDER BY c.created_at DESC
  `).all();

  res.json(campaigns.map(c => ({
    ...c,
    settings: JSON.parse((c as any).settings),
    stats: JSON.parse((c as any).stats),
  })));
});

// POST /api/campaigns
campaignsRouter.post('/', (req: Request, res: Response) => {
  const { name, steps = [], settings = {} } = req.body;
  if (!name) return res.status(400).json({ error: 'name is required' });

  const id = uuidv4();
  const now = new Date().toISOString();
  const db = getDb();

  const defaultSettings = {
    dailyConnectionLimit: 10,
    dailyMessageLimit: 10,
    workingHours: { start: 9, end: 18 },
    workingDays: [1, 2, 3, 4, 5],
    timezone: 'America/New_York',
    ...settings,
  };
  const defaultStats = {
    totalLeads: 0, profilesVisited: 0, connectionsSent: 0,
    connectionsAccepted: 0, messagesSent: 0, replies: 0,
    acceptanceRate: 0, replyRate: 0,
  };

  db.prepare(
    'INSERT INTO campaigns (id, name, status, settings, stats, created_at, updated_at) VALUES (?, ?, ?, ?, ?, ?, ?)'
  ).run(id, name, 'paused', JSON.stringify(defaultSettings), JSON.stringify(defaultStats), now, now);

  // Зберегти кроки
  const insertStep = db.prepare(
    'INSERT INTO campaign_steps (id, campaign_id, type, step_order, config) VALUES (?, ?, ?, ?, ?)'
  );
  (steps as Array<{ type: string; config?: object }>).forEach((step, index) => {
    insertStep.run(uuidv4(), id, step.type, index, JSON.stringify(step.config || {}));
  });

  res.status(201).json({ id, name, status: 'paused' });
});

// GET /api/campaigns/:id
campaignsRouter.get('/:id', (req: Request, res: Response) => {
  const campaign = getDb().prepare('SELECT * FROM campaigns WHERE id = ?').get(req.params.id);
  if (!campaign) return res.status(404).json({ error: 'Campaign not found' });

  const steps = getDb().prepare('SELECT * FROM campaign_steps WHERE campaign_id = ? ORDER BY step_order').all(req.params.id);
  res.json({
    ...(campaign as any),
    settings: JSON.parse((campaign as any).settings),
    stats: JSON.parse((campaign as any).stats),
    steps: steps.map((s: any) => ({ ...s, config: JSON.parse(s.config) })),
  });
});

// PUT /api/campaigns/:id
campaignsRouter.put('/:id', (req: Request, res: Response) => {
  const { name, steps, settings } = req.body;
  const db = getDb();
  const now = new Date().toISOString();

  const updates: string[] = ['updated_at = ?'];
  const values: (string | null)[] = [now];

  if (name) { updates.push('name = ?'); values.push(name); }
  if (settings) { updates.push('settings = ?'); values.push(JSON.stringify(settings)); }

  values.push(req.params.id);
  db.prepare(`UPDATE campaigns SET ${updates.join(', ')} WHERE id = ?`).run(...values);

  if (steps) {
    db.prepare('DELETE FROM campaign_steps WHERE campaign_id = ?').run(req.params.id);
    const insertStep = db.prepare(
      'INSERT INTO campaign_steps (id, campaign_id, type, step_order, config) VALUES (?, ?, ?, ?, ?)'
    );
    (steps as Array<{ type: string; config?: object }>).forEach((step, index) => {
      insertStep.run(uuidv4(), req.params.id, step.type, index, JSON.stringify(step.config || {}));
    });
  }

  res.json({ updated: true });
});

// DELETE /api/campaigns/:id
campaignsRouter.delete('/:id', (req: Request, res: Response) => {
  getDb().prepare('DELETE FROM campaigns WHERE id = ?').run(req.params.id);
  res.json({ deleted: true });
});

// POST /api/campaigns/:id/start
campaignsRouter.post('/:id/start', (req: Request, res: Response) => {
  getDb().prepare("UPDATE campaigns SET status = 'active', updated_at = ? WHERE id = ?")
    .run(new Date().toISOString(), req.params.id);
  res.json({ status: 'active' });
});

// POST /api/campaigns/:id/pause
campaignsRouter.post('/:id/pause', (req: Request, res: Response) => {
  getDb().prepare("UPDATE campaigns SET status = 'paused', updated_at = ? WHERE id = ?")
    .run(new Date().toISOString(), req.params.id);
  res.json({ status: 'paused' });
});

// POST /api/campaigns/:id/resume
campaignsRouter.post('/:id/resume', (req: Request, res: Response) => {
  getDb().prepare("UPDATE campaigns SET status = 'active', updated_at = ? WHERE id = ?")
    .run(new Date().toISOString(), req.params.id);
  res.json({ status: 'active' });
});

// GET /api/campaigns/:id/stats
campaignsRouter.get('/:id/stats', (req: Request, res: Response) => {
  const campaign = getDb().prepare('SELECT stats FROM campaigns WHERE id = ?').get(req.params.id) as any;
  if (!campaign) return res.status(404).json({ error: 'Campaign not found' });
  res.json(JSON.parse(campaign.stats));
});

// POST /api/campaigns/pause-all (emergency)
campaignsRouter.post('/pause-all', (_req: Request, res: Response) => {
  pauseAllCampaigns();
  res.json({ paused: true });
});

import { Router, Request, Response } from 'express';
import { v4 as uuidv4 } from 'uuid';
import multer from 'multer';
import csv from 'csv-parser';
import { Readable } from 'stream';
import { getDb } from '../services/storage';

export const leadsRouter = Router();
const upload = multer({ storage: multer.memoryStorage(), limits: { fileSize: 5 * 1024 * 1024 } });

const LINKEDIN_URL_PATTERN = /^https?:\/\/(www\.)?linkedin\.com\/in\/[\w-]+(\/)?$/i;

// GET /api/leads
leadsRouter.get('/', (req: Request, res: Response) => {
  const { campaignId, status, search, page = '1', limit = '50' } = req.query;
  const offset = (parseInt(page as string) - 1) * parseInt(limit as string);

  let query = 'SELECT * FROM leads WHERE 1=1';
  const params: (string | number)[] = [];

  if (campaignId) { query += ' AND campaign_id = ?'; params.push(campaignId as string); }
  if (status) { query += ' AND status = ?'; params.push(status as string); }
  if (search) {
    query += ' AND (first_name LIKE ? OR last_name LIKE ? OR company LIKE ?)';
    const s = `%${search}%`;
    params.push(s, s, s);
  }

  const total = (getDb().prepare(`SELECT COUNT(*) as count FROM leads WHERE 1=1${query.split('WHERE 1=1')[1].split('ORDER')[0]}`).get(...params) as any).count;
  query += ' ORDER BY created_at DESC LIMIT ? OFFSET ?';
  params.push(parseInt(limit as string), offset);

  const leads = getDb().prepare(query).all(...params);
  res.json({ leads, total, page: parseInt(page as string), limit: parseInt(limit as string) });
});

// POST /api/leads/import
leadsRouter.post('/import', upload.single('file'), async (req: Request, res: Response) => {
  if (!req.file) return res.status(400).json({ error: 'No file uploaded' });

  const results: Array<{ firstName: string; lastName: string; company: string; title: string; profileUrl: string }> = [];
  const errors: string[] = [];

  await new Promise<void>((resolve) => {
    Readable.from(req.file!.buffer)
      .pipe(csv())
      .on('data', (row) => {
        results.push({
          firstName: row.firstName || row.first_name || '',
          lastName: row.lastName || row.last_name || '',
          company: row.company || '',
          title: row.title || '',
          profileUrl: row.profileUrl || row.profile_url || '',
        });
      })
      .on('end', resolve)
      .on('error', (err) => { errors.push(err.message); resolve(); });
  });

  const { campaignId } = req.body;
  const db = getDb();
  let added = 0;
  let skipped = 0;

  for (const row of results) {
    if (!row.profileUrl || !LINKEDIN_URL_PATTERN.test(row.profileUrl)) {
      errors.push(`Invalid LinkedIn URL: ${row.profileUrl}`);
      continue;
    }
    if (!row.firstName) { errors.push(`Missing firstName for ${row.profileUrl}`); continue; }

    const exists = db.prepare('SELECT id FROM leads WHERE profile_url = ?').get(row.profileUrl);
    if (exists) { skipped++; continue; }

    db.prepare(
      'INSERT INTO leads (id, campaign_id, first_name, last_name, company, title, profile_url, created_at, updated_at) VALUES (?, ?, ?, ?, ?, ?, ?, ?, ?)'
    ).run(uuidv4(), campaignId || null, row.firstName, row.lastName, row.company, row.title, row.profileUrl,
      new Date().toISOString(), new Date().toISOString());
    added++;
  }

  res.json({ added, skipped, errors: errors.length, errorDetails: errors.slice(0, 10) });
});

// POST /api/leads (manual add)
leadsRouter.post('/', (req: Request, res: Response) => {
  const { firstName, lastName, company = '', title = '', profileUrl, campaignId } = req.body;
  if (!firstName || !profileUrl) return res.status(400).json({ error: 'firstName and profileUrl are required' });
  if (!LINKEDIN_URL_PATTERN.test(profileUrl)) return res.status(400).json({ error: 'Invalid LinkedIn URL' });

  const exists = getDb().prepare('SELECT id FROM leads WHERE profile_url = ?').get(profileUrl);
  if (exists) return res.status(409).json({ error: 'Lead already exists' });

  const id = uuidv4();
  const now = new Date().toISOString();
  getDb().prepare(
    'INSERT INTO leads (id, campaign_id, first_name, last_name, company, title, profile_url, created_at, updated_at) VALUES (?, ?, ?, ?, ?, ?, ?, ?, ?)'
  ).run(id, campaignId || null, firstName, lastName, company, title, profileUrl, now, now);

  res.status(201).json({ id });
});

// GET /api/leads/:id
leadsRouter.get('/:id', (req: Request, res: Response) => {
  const lead = getDb().prepare('SELECT * FROM leads WHERE id = ?').get(req.params.id);
  if (!lead) return res.status(404).json({ error: 'Lead not found' });
  res.json(lead);
});

// PUT /api/leads/:id
leadsRouter.put('/:id', (req: Request, res: Response) => {
  const allowed = ['first_name', 'last_name', 'company', 'title', 'notes', 'tags', 'campaign_id'];
  const updates = Object.entries(req.body)
    .filter(([k]) => allowed.includes(k))
    .map(([k]) => `${k} = ?`);
  const values = Object.entries(req.body)
    .filter(([k]) => allowed.includes(k))
    .map(([, v]) => typeof v === 'object' ? JSON.stringify(v) : v);

  if (!updates.length) return res.status(400).json({ error: 'No valid fields to update' });

  values.push(new Date().toISOString(), req.params.id);
  getDb().prepare(`UPDATE leads SET ${updates.join(', ')}, updated_at = ? WHERE id = ?`).run(...values);
  res.json({ updated: true });
});

// DELETE /api/leads/:id
leadsRouter.delete('/:id', (req: Request, res: Response) => {
  getDb().prepare('DELETE FROM leads WHERE id = ?').run(req.params.id);
  res.json({ deleted: true });
});

// POST /api/leads/:id/skip
leadsRouter.post('/:id/skip', (req: Request, res: Response) => {
  getDb().prepare("UPDATE leads SET status = 'skipped', updated_at = ? WHERE id = ?")
    .run(new Date().toISOString(), req.params.id);
  res.json({ status: 'skipped' });
});

import { Router, Request, Response } from 'express';
import { getDb } from '../services/storage';

export const analyticsRouter = Router();

// GET /api/analytics/overview
analyticsRouter.get('/overview', (_req: Request, res: Response) => {
  const db = getDb();

  const totalCampaigns = (db.prepare('SELECT COUNT(*) as c FROM campaigns').get() as any).c;
  const activeCampaigns = (db.prepare("SELECT COUNT(*) as c FROM campaigns WHERE status = 'active'").get() as any).c;
  const totalLeads = (db.prepare('SELECT COUNT(*) as c FROM leads').get() as any).c;
  const connectionsSent = (db.prepare("SELECT COUNT(*) as c FROM leads WHERE connection_sent_at IS NOT NULL").get() as any).c;
  const connectionsAccepted = (db.prepare("SELECT COUNT(*) as c FROM leads WHERE connected_at IS NOT NULL").get() as any).c;
  const messagesSent = (db.prepare("SELECT COUNT(*) as c FROM leads WHERE message_sent_at IS NOT NULL").get() as any).c;
  const replies = (db.prepare("SELECT COUNT(*) as c FROM leads WHERE replied_at IS NOT NULL").get() as any).c;

  const acceptanceRate = connectionsSent > 0 ? Math.round((connectionsAccepted / connectionsSent) * 100) : 0;
  const replyRate = messagesSent > 0 ? Math.round((replies / messagesSent) * 100) : 0;

  res.json({
    totalCampaigns, activeCampaigns, totalLeads,
    connectionsSent, connectionsAccepted, messagesSent, replies,
    acceptanceRate, replyRate,
  });
});

// GET /api/analytics/daily?days=30
analyticsRouter.get('/daily', (req: Request, res: Response) => {
  const days = parseInt(req.query.days as string || '30', 10);
  const db = getDb();
  const data = db.prepare(`
    SELECT 
      date,
      connections_sent,
      messages_sent,
      profiles_visited
    FROM daily_tracker
    ORDER BY date DESC
    LIMIT ?
  `).all(days);

  res.json(data.reverse());
});

// GET /api/analytics/campaign/:id
analyticsRouter.get('/campaign/:id', (req: Request, res: Response) => {
  const db = getDb();
  const id = req.params.id;

  const total = (db.prepare('SELECT COUNT(*) as c FROM leads WHERE campaign_id = ?').get(id) as any).c;
  const visited = (db.prepare('SELECT COUNT(*) as c FROM leads WHERE campaign_id = ? AND visited_at IS NOT NULL').get(id) as any).c;
  const sent = (db.prepare('SELECT COUNT(*) as c FROM leads WHERE campaign_id = ? AND connection_sent_at IS NOT NULL').get(id) as any).c;
  const accepted = (db.prepare('SELECT COUNT(*) as c FROM leads WHERE campaign_id = ? AND connected_at IS NOT NULL').get(id) as any).c;
  const messages = (db.prepare('SELECT COUNT(*) as c FROM leads WHERE campaign_id = ? AND message_sent_at IS NOT NULL').get(id) as any).c;
  const replies = (db.prepare('SELECT COUNT(*) as c FROM leads WHERE campaign_id = ? AND replied_at IS NOT NULL').get(id) as any).c;

  res.json({
    totalLeads: total,
    profilesVisited: visited,
    connectionsSent: sent,
    connectionsAccepted: accepted,
    messagesSent: messages,
    replies,
    acceptanceRate: sent > 0 ? Math.round((accepted / sent) * 100) : 0,
    replyRate: messages > 0 ? Math.round((replies / messages) * 100) : 0,
  });
});

import { Router, Request, Response } from 'express';
import { getSetting, setSetting } from '../services/storage';
import { isSessionActive, startManualLogin } from '../services/linkedin';

export const settingsRouter = Router();

const DEFAULT_SETTINGS = {
  dailyConnectionLimit: 10,
  dailyMessageLimit: 10,
  workingHours: { start: 9, end: 18 },
  workingDays: [1, 2, 3, 4, 5],
  timezone: 'America/New_York',
  emailAlert: '',
  myName: '',
};

// GET /api/settings
settingsRouter.get('/', (_req: Request, res: Response) => {
  const stored = getSetting('global_settings');
  const settings = stored ? { ...DEFAULT_SETTINGS, ...JSON.parse(stored) } : DEFAULT_SETTINGS;
  res.json(settings);
});

// PUT /api/settings
settingsRouter.put('/', (req: Request, res: Response) => {
  const stored = getSetting('global_settings');
  const current = stored ? JSON.parse(stored) : DEFAULT_SETTINGS;
  const updated = { ...current, ...req.body };
  setSetting('global_settings', JSON.stringify(updated));
  if (req.body.myName) setSetting('my_name', req.body.myName);
  res.json(updated);
});

// POST /api/settings/login — запустити ручний логін LinkedIn
settingsRouter.post('/login', (_req: Request, res: Response) => {
  // Запускаємо async без await — відповідаємо одразу, процес відбувається у фоні
  startManualLogin().catch(err => {
    console.error('Manual login failed:', err);
  });
  res.json({ status: 'login_started', message: 'Browser opened. Please log in to LinkedIn.' });
});

// GET /api/settings/session — статус LinkedIn сесії
settingsRouter.get('/session', async (_req: Request, res: Response) => {
  try {
    const active = await isSessionActive();
    res.json({
      status: active ? 'active' : 'expired',
      lastChecked: new Date().toISOString(),
    });
  } catch {
    res.json({ status: 'not_connected', lastChecked: new Date().toISOString() });
  }
});

import express from 'express';
import cors from 'cors';
import { initDatabase } from './services/storage';

  initDatabase();
  startCampaignWorker();

    logger.info(`Backend running on port ${PORT}`);
    logger.info(`Frontend URL: ${process.env.FRONTEND_URL || 'http://localhost:5173'}`);
  });
}

start().catch((err) => {
  console.error('Failed to start:', err);
  process.exit(1);
});
